class NegDistribution(object):
    def __init__(self):
        self.mu = 13
        self.sigma = 1

    def sample(self, N):
        samples = np.random.normal(self.mu, self.sigma, N)
        samples.sort()
        return samples

class NegDistribution2(object):
    def __init__(self):
        self.mu = -6
        self.sigma = 2

        samples = np.random.normal(self.mu, self.sigma, N)
